二分查找

思想

二分查找适用于已经有序队列的查找，从中间开始查找，判断大小后取左或者取右，查询效率很高。

具体实现

public class BinarySearch{
    public static void main (String[] args){
        System.out.println("a");
        int[] nums = {
            1,2,3,4,5,6,7,8,9,9,10,11,12,13,15,25,57,678,5345,234432
        };
        System.out.println(Binary(nums,10));
    }
    public static int Binary(int[] nums, int findNum){
        int lo = 0;
        int hi = nums.length - 1;
        int mid = (lo + hi)/2;
        while(lo <= hi){
            if (nums[mid] == findNum) return mid;
            else {
                if (nums[mid] > findNum) hi = mid - 1;
                else lo = mid + 1;
            }
            mid = (lo + hi) / 2;
        }
        return - 1;
    }
}
