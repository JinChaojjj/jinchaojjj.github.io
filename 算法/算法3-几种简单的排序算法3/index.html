算法3-排序算法

i 堆排序

堆排序是比较排序中实现起来较为复杂的一种算法了，其主要步骤分为两步：

  1. 将数组化为大/小顶堆
  2. 进行堆排序

大顶堆

大顶堆堆含义可以简单概括为数组index下标的数比index * 2 + 1和index * 2 + 2的两个数都要大，小顶堆指的就是相较于这两个数都要小。

那么如何化为大顶堆呢，其实很简单，从第一个非叶子节点开始将两个子节点中较大的换上去，若是产生交换，需要对产生交换的节点重新进行一次向下层的比较，依次向上就可以

堆排序

将第一个数和最后一个数进行交换位置，然后对第一个数进行一次堆排序，然后继续交换即可。

代码实现：

public class HeapSort{
    public static void main(String[] args){
        System.out.println("1");
        int[] nums = {
            4,3,5,7,8,5,3,2,5,7,87,6,54,23,45,7,8,76,5,3,2,4,56,7,876
        };
        Heap(nums);
        for(int i = 0; i< nums.length; i++){
            System.out.println(nums[i]);
        }
    }
    public static void Heap(int[] nums){
        for (int i = nums.length-1 ; i>=0;i--){
            change(nums,i,nums.length);
        }
        
        for (int i = nums.length-1 ; i > 0;i--){
            swap(nums,i,0);
            change(nums,0,i-1);
        }

    }

    public static void swap(int[] nums,int i, int j){
        int t = nums[i];
        nums[i] = nums[j];
        nums[j] = t;
    }

    public static void change(int[] nums, int index, int length){
        if (index * 2 + 1 < length && index * 2 + 2 < length){
            if(nums[index] >= nums[index * 2 + 1] && nums[index] > nums[index * 2 + 2]){
                return;
            }else if (nums[index] >= nums[index * 2 + 1] && nums[index] < nums[index * 2 + 2]){
                int t = nums[index];
                nums[index] = nums[index *2+2];
                nums[index * 2 + 2]= t;
                change(nums,index * 2+2 , length);
                return;
            }else if (nums[index] < nums[index * 2+ 1] && nums[index] >= nums[index * 2 + 2]){
                int t = nums[index];
                nums[index] = nums[index * 2 + 1];
                nums[index * 2 + 1] = t;
                change(nums,index*2+1, length);
                return;
            }else{
                if (nums[index * 2 + 1] > nums[index * 2 + 2]){
                    int t = nums[index];
                    nums[index] = nums[index * 2 + 1];
                    nums[index * 2 + 1] = t;
                    change(nums,index * 2 + 1, length);
                    return;
                }else{
                    int t = nums[index];
                    nums[index] = nums[index *2+2];
                    nums[index * 2 + 2]= t;
                    change(nums,index * 2+ 2, length);
                    return;
                }
            }
        }
        if (index * 2 + 1 < length && index * 2 + 2 >= length){
            if (nums[index] >= nums[index * 2 + 1]){
                return;
            }else{
                int t = nums[index];
                nums[index] = nums[index * 2 + 1];
                nums[index * 2 + 1] = t;
                change(nums, index * 2 + 1, length);
                return;
            }
        }
        if (index * 2 + 1 >= length){
            return;
        }
    }
}
